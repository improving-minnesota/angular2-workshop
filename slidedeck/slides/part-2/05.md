## Typescript

### Interface Constructor

```
export interface DogData {
  name?: string;
  breed?: string;
  age?: number;
}

export class Dog {
  name: string;
  breed: string;
  age: number;

  constructor(data: DogData) {
    this.name = data.name;
    this.breed = data.breed;
    this.age = data.age;
  }
}
```

```
let dogData = {name: 'Bear', breed: 'Border Collie', age: 4};
let dog = new Dog(dogData);
```
